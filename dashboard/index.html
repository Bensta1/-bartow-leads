fetch('records.json') .then(r => r.json()) .then(data => { const records = data.records || []; document.getElementById('count').textContent = `Total Leads: ${records.length} | Last updated: ${data.fetched_at || 'Never'}`; const body = document.getElementById('leads-body'); records.forEach(lead => { body.innerHTML += ` ${lead.name || ''} ${lead.address || ''} ${lead.case_number || ''} ${lead.date || ''} ${lead.doc_type || ''} `; }); }) .catch(() => { document.getElementById('count').textContent = 'No data yet — run the scraper!'; });
